# Custom Memory

Memory backends provide persistent vector-based memory for Pandora. This guide walks you through creating one.

## Example: Postgres + pgvector backend

Let's build a memory backend using Postgres with the pgvector extension.

### Step 1: Install dependencies

```bash
bun add pg
```

### Step 2: Create the file

Create `packages/pandora/src/memory/postgres.ts`:

```typescript
import {
  defineMemory,
  generateId,
  createEmbeddingModel,
  type IMemoryProvider,
  type IEpisodicMemory,
  type ISemanticMemory,
  type Episode,
  type Fact,
  type MemoryChunk,
  type MemorySearchResults,
  type MemoryConfig,
} from "@pandora/core";
import { embedMany } from "ai";
import { Pool } from "pg";

interface PostgresConfig extends MemoryConfig {
  connectionString: string;
}

class PostgresMemoryProvider implements IMemoryProvider {
  private pool: Pool;
  private embeddingModel: string;
  private apiKey: string;

  constructor(pool: Pool, embeddingModel: string, apiKey: string) {
    this.pool = pool;
    this.embeddingModel = embeddingModel;
    this.apiKey = apiKey;
  }

  static async create(config: PostgresConfig): Promise<PostgresMemoryProvider> {
    const pool = new Pool({ connectionString: config.connectionString });

    // Create tables with pgvector
    await pool.query(`
      CREATE EXTENSION IF NOT EXISTS vector;

      CREATE TABLE IF NOT EXISTS memory_episodes (
        id TEXT PRIMARY KEY,
        content TEXT NOT NULL,
        conversation_id TEXT,
        channel_name TEXT,
        user_id TEXT,
        timestamp INTEGER NOT NULL,
        importance REAL DEFAULT 0.5,
        tags TEXT,
        embedding vector(1536)
      );

      CREATE TABLE IF NOT EXISTS memory_facts (
        id TEXT PRIMARY KEY,
        content TEXT NOT NULL,
        category TEXT NOT NULL,
        created_at INTEGER NOT NULL,
        updated_at INTEGER NOT NULL,
        source TEXT,
        confidence REAL DEFAULT 1.0,
        embedding vector(1536)
      );

      CREATE INDEX IF NOT EXISTS idx_episodes_embedding
        ON memory_episodes USING ivfflat (embedding vector_cosine_ops);

      CREATE INDEX IF NOT EXISTS idx_facts_embedding
        ON memory_facts USING ivfflat (embedding vector_cosine_ops);
    `);

    return new PostgresMemoryProvider(
      pool,
      config.embeddingModel ?? "openai/text-embedding-3-small",
      config.apiKey ?? ""
    );
  }

  private async embed(texts: string[]): Promise<number[][]> {
    if (texts.length === 0) return [];
    const model = createEmbeddingModel(this.embeddingModel, this.apiKey);
    const { embeddings } = await embedMany({ model, values: texts });
    return embeddings;
  }

  readonly episodic: IEpisodicMemory = {
    addEpisode: async (episode) => {
      const id = generateId();
      const [embedding] = await this.embed([episode.content]);

      await this.pool.query(
        `INSERT INTO memory_episodes
         (id, content, conversation_id, channel_name, user_id, timestamp, importance, tags, embedding)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)`,
        [
          id,
          episode.content,
          episode.conversationId,
          episode.channelName,
          episode.userId,
          episode.timestamp,
          episode.importance,
          JSON.stringify(episode.tags),
          JSON.stringify(embedding),
        ]
      );

      return id;
    },

    searchEpisodes: async (query, opts = {}) => {
      const { limit = 10, minScore = 0.5 } = opts;
      const [queryVector] = await this.embed([query]);

      const result = await this.pool.query(
        `SELECT id, content, conversation_id, timestamp,
                1 - (embedding <=> $1::vector) as score
         FROM memory_episodes
         WHERE 1 - (embedding <=> $1::vector) >= $2
         ORDER BY embedding <=> $1::vector
         LIMIT $3`,
        [JSON.stringify(queryVector), minScore, limit]
      );

      return result.rows.map((row) => ({
        chunkId: row.id,
        content: row.content,
        parentId: row.id,
        parentType: "episode" as const,
        score: row.score,
        chunkIndex: 0,
        timestamp: row.timestamp,
        conversationId: row.conversation_id,
      }));
    },

    getEpisode: async (id) => {
      const result = await this.pool.query(
        `SELECT * FROM memory_episodes WHERE id = $1`,
        [id]
      );
      if (result.rows.length === 0) return null;

      const row = result.rows[0];
      return {
        id: row.id,
        content: row.content,
        conversationId: row.conversation_id,
        channelName: row.channel_name,
        userId: row.user_id,
        timestamp: row.timestamp,
        importance: row.importance,
        tags: row.tags ? JSON.parse(row.tags) : [],
      };
    },

    deleteEpisode: async (id) => {
      await this.pool.query(`DELETE FROM memory_episodes WHERE id = $1`, [id]);
    },

    deleteEpisodesForConversation: async (conversationId) => {
      const result = await this.pool.query(
        `DELETE FROM memory_episodes WHERE conversation_id = $1`,
        [conversationId]
      );
      return result.rowCount ?? 0;
    },
  };

  readonly semantic: ISemanticMemory = {
    upsertFact: async (fact) => {
      const id = generateId();
      const now = Math.floor(Date.now() / 1000);
      const [embedding] = await this.embed([fact.content]);

      await this.pool.query(
        `INSERT INTO memory_facts
         (id, content, category, created_at, updated_at, source, confidence, embedding)
         VALUES ($1, $2, $3, $4, $5, $6, $7, $8)`,
        [
          id,
          fact.content,
          fact.category,
          now,
          now,
          fact.source,
          fact.confidence,
          JSON.stringify(embedding),
        ]
      );

      return id;
    },

    searchFacts: async (query, opts = {}) => {
      const { limit = 10, minScore = 0.5, category } = opts;
      const [queryVector] = await this.embed([query]);

      let sql = `
        SELECT id, content, category,
               1 - (embedding <=> $1::vector) as score
        FROM memory_facts
        WHERE 1 - (embedding <=> $1::vector) >= $2
      `;
      const params: (string | number)[] = [JSON.stringify(queryVector), minScore];

      if (category) {
        sql += ` AND category = $3`;
        params.push(category);
      }

      sql += ` ORDER BY embedding <=> $1::vector LIMIT $${params.length + 1}`;
      params.push(limit);

      const result = await this.pool.query(sql, params);

      return result.rows.map((row) => ({
        chunkId: row.id,
        content: row.content,
        parentId: row.id,
        parentType: "fact" as const,
        score: row.score,
        chunkIndex: 0,
        category: row.category,
      }));
    },

    getFact: async (id) => {
      const result = await this.pool.query(
        `SELECT * FROM memory_facts WHERE id = $1`,
        [id]
      );
      if (result.rows.length === 0) return null;

      const row = result.rows[0];
      return {
        id: row.id,
        content: row.content,
        category: row.category,
        createdAt: row.created_at,
        updatedAt: row.updated_at,
        source: row.source,
        confidence: row.confidence,
      };
    },

    deleteFact: async (id) => {
      await this.pool.query(`DELETE FROM memory_facts WHERE id = $1`, [id]);
    },
  };

  async search(query: string, opts = {}): Promise<MemorySearchResults> {
    const { limit = 10, minScore = 0.5 } = opts;

    const [episodes, facts] = await Promise.all([
      this.episodic.searchEpisodes(query, { limit, minScore }),
      this.semantic.searchFacts(query, { limit, minScore }),
    ]);

    return { episodes, facts };
  }

  async close(): Promise<void> {
    await this.pool.end();
  }
}

export default defineMemory({
  type: "postgres",
  create: (config) => PostgresMemoryProvider.create(config as PostgresConfig),
});
```

### Step 3: Add to config

```jsonc
"memory": {
  "type": "postgres",
  "connectionString": "postgresql://user:pass@localhost:5432/pandora",
  "embeddingModel": "openai/text-embedding-3-small"
}
```

### Step 4: Done!

Restart Pandora. Memory is now stored in Postgres with pgvector.

## The IMemoryProvider interface

Your memory backend must implement:

```typescript
interface IMemoryProvider {
  readonly episodic: IEpisodicMemory | null;
  readonly semantic: ISemanticMemory | null;

  search(query: string, opts?: { limit?: number; minScore?: number }): Promise<MemorySearchResults>;
  close(): Promise<void>;
}
```

You can implement just one memory type by setting the other to `null`.

## IEpisodicMemory interface

Automatic interaction logging:

```typescript
interface IEpisodicMemory {
  addEpisode(episode: Omit<Episode, "id">): Promise<string>;
  searchEpisodes(query: string, opts?: SearchOptions): Promise<MemoryChunk[]>;
  getEpisode(id: string): Promise<Episode | null>;
  deleteEpisode(id: string): Promise<void>;
  deleteEpisodesForConversation(conversationId: string): Promise<number>;
}
```

| Method | Description |
|--------|-------------|
| `addEpisode` | Store an interaction (gateway calls this automatically) |
| `searchEpisodes` | Find relevant past interactions |
| `getEpisode` | Get full episode by ID |
| `deleteEpisode` | Delete a specific episode |
| `deleteEpisodesForConversation` | Clean up when conversation is deleted |

---

## ISemanticMemory interface

Agent-controlled facts and knowledge:

```typescript
interface ISemanticMemory {
  upsertFact(fact: Omit<Fact, "id" | "createdAt" | "updatedAt">): Promise<string>;
  searchFacts(query: string, opts?: SearchOptions): Promise<MemoryChunk[]>;
  getFact(id: string): Promise<Fact | null>;
  deleteFact(id: string): Promise<void>;
}
```

| Method | Description |
|--------|-------------|
| `upsertFact` | Create or update a fact (dedupe by similarity) |
| `searchFacts` | Find relevant facts |
| `getFact` | Get full fact by ID |
| `deleteFact` | Delete a fact |

## Data types

### Episode

```typescript
interface Episode {
  id: string;
  content: string;              // "User: ... Assistant: ..."
  conversationId?: string;
  channelName?: string;
  userId?: string;
  timestamp: number;            // Unix seconds
  importance: number;           // 0-1
  tags: string[];
}
```

### Fact

```typescript
interface Fact {
  id: string;
  content: string;              // The fact text
  category: string;             // user_preference, knowledge, instruction
  createdAt: number;            // Unix seconds
  updatedAt: number;
  source?: string;
  confidence: number;           // 0-1
}
```

### MemoryChunk

```typescript
interface MemoryChunk {
  chunkId: string;
  content: string;
  parentId: string;
  parentType: "episode" | "fact";
  score: number;                // 0-1 similarity
  chunkIndex: number;
  timestamp?: number;           // Episodes only
  conversationId?: string;      // Episodes only
  category?: string;            // Facts only
}
```

## Tips

- **Vector indexing** — Use specialized indices (IVFFlat, HNSW) for fast similarity search
- **Embedding batching** — Use `embedMany()` for efficiency
- **Chunking** — For long content, split into overlapping chunks before embedding
- **Deduplication** — Check similarity before inserting facts to avoid duplicates
- **Cleanup** — Episodes should be deleted with their parent conversation
